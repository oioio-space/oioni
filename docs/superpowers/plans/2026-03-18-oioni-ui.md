```go
// ui/gui/draw.go — shared drawing primitives for gui widgets
package gui

import (
	"image"
	"image/color"
	"math"

	"github.com/oioio-space/oioni/ui/canvas"
)

// DrawRoundedRect draws a rounded rectangle outline (fill=false) or filled shape (fill=true).
// radius is the corner radius in pixels. col must be canvas.Black or canvas.White.
// Uses Midpoint Circle Algorithm on corners, straight edges between corners.
func DrawRoundedRect(c *canvas.Canvas, r image.Rectangle, radius int, fill bool, col color.Color) {
	if r.Empty() {
		return
	}
	if radius < 0 {
		radius = 0
	}
	// Clamp radius so it fits the rect
	maxR := min(r.Dx(), r.Dy()) / 2
	if radius > maxR {
		radius = maxR
	}

	x0, y0 := r.Min.X, r.Min.Y
	x1, y1 := r.Max.X-1, r.Max.Y-1

	// Corner centers
	tlX, tlY := x0+radius, y0+radius
	trX, trY := x1-radius, y0+radius
	blX, blY := x0+radius, y1-radius
	brX, brY := x1-radius, y1-radius

	if fill {
		// Fill: for each y, draw horizontal span
		for y := y0; y <= y1; y++ {
			// Determine x extent at this y
			var xLeft, xRight int
			// Which vertical band are we in?
			switch {
			case y < tlY: // top corners
				dy := float64(tlY - y)
				dx := int(math.Sqrt(float64(radius*radius) - dy*dy))
				xLeft = tlX - dx
				xRight = trX + dx
			case y > blY: // bottom corners
				dy := float64(y - blY)
				dx := int(math.Sqrt(float64(radius*radius) - dy*dy))
				xLeft = blX - dx
				xRight = brX + dx
			default: // middle band
				xLeft = x0
				xRight = x1
			}
			for x := xLeft; x <= xRight; x++ {
				c.SetPixel(x, y, col)
			}
		}
	} else {
		// Outline: draw 4 straight edges + 4 corner arcs

		// Top and bottom edges
		for x := tlX; x <= trX; x++ {
			c.SetPixel(x, y0, col)
			c.SetPixel(x, y1, col)
		}
		// Left and right edges
		for y := tlY; y <= blY; y++ {
			c.SetPixel(x0, y, col)
			c.SetPixel(x1, y, col)
		}

		// Corner arcs using integer midpoint circle
		if radius > 0 {
			drawCornerArc(c, tlX, tlY, radius, 2, col) // top-left:  quadrant 2 (x-, y-)
			drawCornerArc(c, trX, trY, radius, 1, col) // top-right: quadrant 1 (x+, y-)
			drawCornerArc(c, blX, blY, radius, 3, col) // bot-left:  quadrant 3 (x-, y+)
			drawCornerArc(c, brX, brY, radius, 4, col) // bot-right: quadrant 4 (x+, y+)
		}
	}
}

// drawCornerArc draws one quadrant of a circle arc using Bresenham's midpoint algorithm.
// quad: 1=top-right, 2=top-left, 3=bot-left, 4=bot-right.
func drawCornerArc(c *canvas.Canvas, cx, cy, r, quad int, col color.Color) {
	x, y := 0, r
	d := 1 - r
	for x <= y {
		switch quad {
		case 1:
			c.SetPixel(cx+x, cy-y, col)
			c.SetPixel(cx+y, cy-x, col)
		case 2:
			c.SetPixel(cx-x, cy-y, col)
			c.SetPixel(cx-y, cy-x, col)
		case 3:
			c.SetPixel(cx-x, cy+y, col)
			c.SetPixel(cx-y, cy+x, col)
		case 4:
			c.SetPixel(cx+x, cy+y, col)
			c.SetPixel(cx+y, cy+x, col)
		}
		x++
		if d < 0 {
			d += 2*x + 1
		} else {
			y--
			d += 2*(x-y) + 1
		}
	}
}
```

**Update `Run()`** to handle `renderCh`, `wakeCh`, and idle sleep:

```go
func (nav *Navigator) Run(ctx context.Context, events <-chan touch.TouchEvent) {
	var swipePt *touch.TouchPoint
	var swipeTimer *time.Timer

	timerCh := func() <-chan time.Time {
		if swipeTimer != nil {
			return swipeTimer.C
		}
		return nil
	}
	flush := func() {
		if swipePt != nil {
			nav.handleTouch(*swipePt)
			swipePt = nil
		}
		if swipeTimer != nil {
			swipeTimer.Stop()
			swipeTimer = nil
		}
	}

	// Idle sleep state
	sleeping := false
	var idleTimer *time.Timer
	var idleTimerCh <-chan time.Time

	// drainTimer safely stops a timer and drains its channel to avoid stale fires.
	drainTimer := func(t *time.Timer) {
		if !t.Stop() {
			select {
			case <-t.C:
			default:
			}
		}
	}

	resetIdle := func() {
		if nav.idleTimeout <= 0 {
			return
		}
		if idleTimer != nil {
			drainTimer(idleTimer)
		}
		idleTimer = time.NewTimer(nav.idleTimeout)
		idleTimerCh = idleTimer.C
	}
	if nav.idleTimeout > 0 {
		resetIdle()
	}

	touchEvents := events
	if touchEvents == nil {
		touchEvents = make(chan touch.TouchEvent) // never fires, avoids nil-channel block
	}

	for {
		select {
		case <-ctx.Done():
			flush()
			return

		case <-idleTimerCh:
			if !sleeping {
				sleeping = true
				_ = nav.display.Sleep()
			}
			idleTimerCh = nil // don't re-fire until reset

		case <-nav.wakeCh:
			if sleeping {
				sleeping = false
				_ = nav.display.Init(epd.ModeFull)
				nav.RenderWith(true) //nolint:errcheck
			}
			resetIdle()

		case <-nav.renderCh:
			nav.Render() //nolint:errcheck

		case <-timerCh():
			if swipePt != nil {
				nav.handleTouch(*swipePt)
				swipePt = nil
			}
			swipeTimer = nil

		case ev, ok := <-touchEvents:
			if !ok {
				flush()
				return
			}
			// Wake display if sleeping
			if sleeping {
				sleeping = false
				_ = nav.display.Init(epd.ModeFull)
				nav.RenderWith(true) //nolint:errcheck
			}
			// Reset idle timer on every touch
			resetIdle()

			for _, pt := range ev.Points {
				if swipePt == nil {
					cp := pt
					swipePt = &cp
					swipeTimer = time.NewTimer(300 * time.Millisecond)
					continue
				}
				swipeTimer.Stop()
				swipeTimer = nil
				firstPt := *swipePt
				swipePt = nil
				dx := int(pt.X) - int(firstPt.X)
				dy := int(pt.Y) - int(firstPt.Y)
				adx, ady := dx, dy
				if adx < 0 { adx = -adx }
				if ady < 0 { ady = -ady }
				const threshold = 30
				if adx >= ady && adx > threshold {
					if dx < 0 {
						// Swipe left: route to hScrollable or Pop
						routed := false
						if len(nav.stack) > 0 {
							for _, w := range nav.stack[len(nav.stack)-1].Widgets {
								if hs, ok := w.(hScrollable); ok {
									hs.ScrollH(-1)
									routed = true
									break
								}
							}
						}
						if !routed {
							nav.Pop() //nolint
						}
					} else {
						// Swipe right: route to hScrollable
						if len(nav.stack) > 0 {
							for _, w := range nav.stack[len(nav.stack)-1].Widgets {
								if hs, ok := w.(hScrollable); ok {
									hs.ScrollH(+1)
									break
								}
							}
						}
					}
				} else if ady > adx && ady > threshold {
					if len(nav.stack) > 0 {
						for _, w := range nav.stack[len(nav.stack)-1].Widgets {
							if s, ok := w.(scrollable); ok {
								if dy < 0 { s.Scroll(-1) } else { s.Scroll(1) }
								break
							}
						}
					}
				} else {
					nav.handleTouch(firstPt)
					nav.handleTouch(pt)
				}
			}
			nav.Render() //nolint:errcheck
		}
	}
}
```

```go
	nb.SetDirty()

	b := nb.Bounds()
	if b.Empty() {
		return
	}

	f := canvas.EmbeddedFont(8)

	// Build breadcrumb string
	text := strings.Join(nb.path, " › ")

	// Truncate if too wide
	maxW := b.Dx() - 4

		// Truncate: show "… › <last>"

			}

	if f != nil {
		c.DrawText(b.Min.X+2, b.Min.Y+2, text, f, canvas.Black)
	}

	// Separator line at bottom
```

```go
// ui/gui/widget_carousel.go — IconCarousel: horizontal icon category picker
package gui

import (
	"image"
	"time"

	"github.com/oioio-space/oioni/ui/canvas"
	"github.com/oioio-space/oioni/drivers/touch"
)

const (
	carouselButtonSize = 80  // px per button
	carouselGap        = 6   // px gap between buttons
	carouselLeading    = 13  // px leading indent
	carouselPagHeight  = 8   // px for pagination row at bottom
	carouselContentH   = 80  // px for button area
)

// CarouselItem describes one button in the IconCarousel.
type CarouselItem struct {
	Icon  Icon
	Label string
	OnTap func() // called after tap feedback delay; may be nil
}

// IconCarousel is a horizontally scrollable row of icon+label category buttons.
// It implements hScrollable for Navigator swipe routing.
// It implements Touchable for tap handling.
//
// hScrollable constraint: must be a direct member of Scene.Widgets (not wrapped in a layout
// container) for Navigator.Run() horizontal swipe routing to find it.
type IconCarousel struct {
	BaseWidget
	items    []CarouselItem
	index    int  // leftmost visible button index (snap position)
	pressed  int  // index of currently pressed button (-1 = none)
	tapDelay time.Duration // override for tests; 0 = use default 100ms
}

// NewIconCarousel creates an IconCarousel with the given items.
func NewIconCarousel(items []CarouselItem) *IconCarousel {
	c := &IconCarousel{
		items:   items,
		pressed: -1,
	}
	c.SetDirty()
	return c
}

func (c *IconCarousel) PreferredSize() image.Point { return image.Pt(206, 88) }
func (c *IconCarousel) MinSize() image.Point       { return image.Pt(100, 60) }

// Index returns the current snap position (leftmost visible button index).
func (c *IconCarousel) Index() int { return c.index }

// SetIndex restores the scroll position.
func (c *IconCarousel) SetIndex(i int) {
	c.index = clamp(i, 0, max(0, len(c.items)-1))
	c.SetDirty()
}

// ScrollH implements hScrollable.
// delta=-1: scroll right (next item enters from right, triggered by swipe-left).
// delta=+1: scroll left (previous item, triggered by swipe-right).
func (c *IconCarousel) ScrollH(delta int) {
	c.index = clamp(c.index-delta, 0, max(0, len(c.items)-1))
	c.SetDirty()
}

// HandleTouch implements Touchable. Non-blocking: schedules OnTap via time.AfterFunc.
func (c *IconCarousel) HandleTouch(pt touch.TouchPoint) bool {
	b := c.Bounds()
	if b.Empty() || len(c.items) == 0 {
		return false
	}

	// Logical coordinates: pt is already logical (Navigator transforms before calling)
	// Determine which button was tapped
	x := int(pt.X) - b.Min.X
	buttonIdx := -1
	bx := carouselLeading
	for i := c.index; i < len(c.items); i++ {
		if x >= bx && x < bx+carouselButtonSize {
			buttonIdx = i
			break
		}
		bx += carouselButtonSize + carouselGap
		if bx >= b.Dx() {
			break
		}
	}

	if buttonIdx < 0 || buttonIdx >= len(c.items) {
		return false
	}

	// Tap feedback: invert button, then fire OnTap after delay
	c.pressed = buttonIdx
	c.SetDirty()
	delay := c.tapDelay
	if delay <= 0 {
		delay = 100 * time.Millisecond
	}
	item := c.items[buttonIdx]
	time.AfterFunc(delay, func() {
		c.pressed = -1
		c.SetDirty()
		if item.OnTap != nil {
			item.OnTap()
		}
	})
	return true
}

// Draw renders the carousel buttons, pagination dots, and scroll arrows.
func (c *IconCarousel) Draw(cv *canvas.Canvas) {
	b := c.Bounds()
	if b.Empty() {
		return
	}

	f8 := canvas.EmbeddedFont(8)
	contentRect := image.Rect(b.Min.X, b.Min.Y, b.Max.X, b.Max.Y-carouselPagHeight)

	// Draw visible buttons
	bx := b.Min.X + carouselLeading
	for i := c.index; i < len(c.items); i++ {
		btnRect := image.Rect(bx, b.Min.Y, bx+carouselButtonSize, b.Min.Y+carouselContentH)
		if btnRect.Min.X >= b.Max.X {
			break
		}
		// Clip to content area
		if btnRect.Max.X > b.Max.X {
			btnRect.Max.X = b.Max.X
		}
		_ = contentRect

		inverted := c.pressed == i
		bg := canvas.White
		fg := canvas.Black
		if inverted {
			bg, fg = fg, bg
		}

		// Fill background
		DrawRoundedRect(cv, btnRect, 4, true, bg)
		// Draw border
		DrawRoundedRect(cv, btnRect, 4, false, fg)

		// Draw icon (centered in top 60px of button)
		iconRect := image.Rect(
			btnRect.Min.X+(carouselButtonSize-32)/2,
			btnRect.Min.Y+(carouselContentH-32)/2-8,
			btnRect.Min.X+(carouselButtonSize+32)/2,
			btnRect.Min.Y+(carouselContentH-32)/2-8+32,
		)
		if iconRect.Min.X >= b.Min.X && iconRect.Max.X <= b.Max.X {
			c.items[i].Icon.Draw(cv, iconRect)
		}

		// Draw label
		if f8 != nil && c.items[i].Label != "" {
			lw := textWidth(c.items[i].Label, f8)
			lx := btnRect.Min.X + (carouselButtonSize-lw)/2
			ly := btnRect.Min.Y + carouselContentH - f8.LineHeight() - 2
			if lx >= b.Min.X && lx < b.Max.X {
				cv.DrawText(lx, ly, c.items[i].Label, f8, fg)
			}
		}

		bx += carouselButtonSize + carouselGap
	}

	// Pagination dots
	// Confirmed canvas API: DrawCircle(cx, cy, radius int, col color.Color, filled bool)
	if len(c.items) > 1 {
		dotY := b.Max.Y - carouselPagHeight/2
		dotSpacing := 6
		totalW := len(c.items)*dotSpacing - 1
		startX := b.Min.X + (b.Dx()-totalW)/2
		for i := range c.items {
			dx := startX + i*dotSpacing
			if i == c.index {
				cv.DrawCircle(dx, dotY, 2, canvas.Black, true)
			} else {
				cv.DrawCircle(dx, dotY, 2, canvas.Black, false)
			}
		}
	}

	// Scroll arrows
	if c.index > 0 {
		cv.DrawText(b.Min.X, b.Min.Y+carouselContentH/2-4, "‹", f8, canvas.Black)
	}
	if c.index < len(c.items)-1 {
		// Check if last item is out of view

			cv.DrawText(b.Max.X-8, b.Min.Y+carouselContentH/2-4, "›", f8, canvas.Black)
		}
	}
}
```

```go
// SetButtons replaces the button set and marks the widget dirty.
func (s *ActionSidebar) SetButtons(buttons ...SidebarButton) {
	s.buttons = append([]SidebarButton(nil), buttons...)
	s.SetDirty()
}

func (s *ActionSidebar) PreferredSize() image.Point { return image.Pt(44, 122) }
func (s *ActionSidebar) MinSize() image.Point       { return image.Pt(44, 40) }

// HandleTouch routes a tap to the button at the touch's Y position.
func (s *ActionSidebar) HandleTouch(pt touch.TouchPoint) bool {
	b := s.Bounds()
	if b.Empty() || len(s.buttons) == 0 {
		return false
	}
	y := int(pt.Y) - b.Min.Y
	cellH := b.Dy() / len(s.buttons)
	if cellH <= 0 {
		return false
	}
	idx := y / cellH
	if idx < 0 {
		idx = 0
	}
	if idx >= len(s.buttons) {
		idx = len(s.buttons) - 1
	}
	if s.buttons[idx].OnTap != nil {
		s.buttons[idx].OnTap()
	}
	return true
}

// Draw renders the sidebar: left separator line + equally distributed icon buttons.
func (s *ActionSidebar) Draw(c *canvas.Canvas) {
	b := s.Bounds()
	if b.Empty() {
		return
	}

	// Left separator line
	c.DrawLine(b.Min.X, b.Min.Y, b.Min.X, b.Max.Y-1, canvas.Black)

	if len(s.buttons) == 0 {
		return
	}

	cellH := b.Dy() / len(s.buttons)
	for i, btn := range s.buttons {
		cellY := b.Min.Y + i*cellH
		cellRect := image.Rect(b.Min.X+1, cellY, b.Max.X, cellY+cellH)

		// Separator between buttons (not after last)
		if i > 0 {
			c.DrawLine(b.Min.X+1, cellY, b.Max.X-1, cellY, canvas.Black)
		}

		// Icon centered in cell
		iconSize := 24
		iconX := cellRect.Min.X + (cellRect.Dx()-iconSize)/2
		iconY := cellRect.Min.Y + (cellRect.Dy()-iconSize)/2
		iconRect := image.Rect(iconX, iconY, iconX+iconSize, iconY+iconSize)
		if !iconRect.Empty() {
			btn.Icon.Draw(c, iconRect)
		}
	}
}
```

```go
// cmd/oioni/ui/keepalive.go — 24h e-paper keep-alive (Waveshare hardware requirement)
//
// Per Waveshare 2.13" Touch e-Paper HAT hardware manual:
// "If the screen is not refreshed for a long time, it will cause permanent damage to the screen."
// The maximum idle time between refreshes is approximately 24 hours.
//
// StartKeepAlive runs a goroutine that wakes the Navigator every 24 hours,
// triggering a full refresh even without user interaction.
package ui

import (
	"context"
	"time"

	"github.com/oioio-space/oioni/ui/gui"
)

const keepAliveInterval = 24 * time.Hour

// StartKeepAlive starts a background goroutine that calls nav.Wake() every 24 hours.
// The goroutine stops when ctx is cancelled.
func StartKeepAlive(ctx context.Context, nav *gui.Navigator) {
	go func() {
		t := time.NewTicker(keepAliveInterval)
		defer t.Stop()
		for {
			select {
			case <-ctx.Done():
				return
			case <-t.C:
				nav.Wake()
			}
		}
	}()
}
```

```go
	epdSPIDevice = "/dev/spidev0.0"
	epdSPISpeed  = uint32(4_000_000)
	epdPinRST    = 17
	epdPinDC     = 25
	epdPinCS     = 8
	epdPinBUSY   = 24
	touchDevice  = "/dev/i2c-1"
	touchAddr    = 0x14
	touchPinTRST = 22
	touchPinINT  = 27

	idleTimeout = 60 * time.Second
)

type epaperState struct {
	nav    *gui.Navigator
	status *gui.StatusBar
	td     *touch.Detector
	cancel context.CancelFunc

func startEPaper(ctx context.Context) *epaperState {
	d, err := epd.New(epd.Config{
		SPIDevice: epdSPIDevice,
		SPISpeed:  epdSPISpeed,
		PinRST:    epdPinRST,
		PinDC:     epdPinDC,
		PinCS:     epdPinCS,
		PinBUSY:   epdPinBUSY,
	})
	if err != nil {
		log.Printf("epaper: display unavailable: %v", err)
		return nil
	}

	td, err := touch.New(touch.Config{
		I2CDevice: touchDevice,
		I2CAddr:   uint16(touchAddr),
		PinTRST:   touchPinTRST,
		PinINT:    touchPinINT,
	})
	if err != nil {
		log.Printf("epaper: touch unavailable: %v", err)
		_ = d.Close()
		return nil
	}

	guiCtx, cancel := context.WithCancel(ctx)
	tc, err := td.Start(guiCtx)
	if err != nil {
		log.Printf("epaper: touch start failed: %v", err)
		cancel()
		_ = d.Close()
		return nil
	}

	nav := gui.NewNavigatorWithIdle(d, idleTimeout)
	status := gui.NewStatusBar("", "")

	home := oioniui.NewHomeScene(nav, status)
	if err := nav.Push(home); err != nil {
		log.Printf("epaper: initial render failed: %v", err)
		cancel()
		_ = d.Sleep()
		_ = d.Close()
		return nil
	}

	oioniui.StartKeepAlive(guiCtx, nav)

	go func() {
		defer func() {
			_ = td.Close()
			_ = d.Sleep()
			_ = d.Close()
		}()
		nav.Run(guiCtx, tc)
	}()

	return &epaperState{nav: nav, status: status, td: td, cancel: cancel}

// UpdateStatus updates the status bar text and triggers a re-render via RequestRender.
func (e *epaperState) UpdateStatus(left, right string) {
	if e == nil {
		return
	}
	e.status.SetLeft(left)
	e.status.SetRight(right)
	e.nav.RequestRender()
}

// Close cancels the GUI goroutine and releases hardware resources.
func (e *epaperState) Close() {
	if e == nil {
		return
	}
	e.cancel()
}
```

## Design notes

**NavBar static path (HomeScene):** `navbar := gui.NewNavBar("Home")` is set once at construction. This is intentional — the Home scene's breadcrumb never changes. There is no `OnEnter` update needed. For category pages, `gui.NewNavBar("Home", title)` is also set at construction (each category scene is a new instance on each Push). The `Scene.Title` field is metadata only; Navigator does not read it.

**hScrollable dual-listing in Scene.Widgets:** `carousel` appears twice in `HomeScene.Widgets` — once inside the `root` HBox layout (for rendering), and once directly in the slice (for `hScrollable` routing). This is the required pattern per the spec. The root layout container renders carousel at its allocated bounds; the direct slice entry makes it findable by Navigator's horizontal swipe scan (which only checks top-level `Widgets`, not children of layout containers).
